1 #Windows System Information Tool with Python
2 
3 Windows sistem bilgilerini Python ile alırsınız. Aslında zaten var olan bir kütüphanenin daha da basitleştirilmiş halidir.
4 
5 Kullanmadan önce eğer gerekli kütüphane yok ise onu kurmanız gerekli. **syspy**, gösterdiği verileri **psutil** ile sağlar.
6 
7 ~~~~{.python}
8 from syspy import CPU
9 from syspy import Disk
10 from syspy import Info
11 from syspy import Memory
12 from syspy import Network
13 
14 cpu = CPU()
15 
16 print(cpu.getCpuTimes())
17 
18 disk = Disk()
19 
20 a,b,c,d,e,f,g = disk.getDiskPartitions()
21 
22 print("Part Device: {}\nUsage Total: {}\nUsage Used: {}\nUsage Free: {}" \
23  "\nUsage Percent: {}\nPart Type: {}\nPart Mountpoint: {}".format(a,b,c,d,e,f,g))
24 
25 info = Info()
26 
27 print(info.getBootTime())
28 
29 memory = Memory()
30 
31 a,b,c,d,e,f = memory.getSwapMemory()
32 print("Total: {0} Used: {1} Free: {2} Percent: {3} Sin: {4} Sout: {5}".format(a,b,c,d,e,f))
